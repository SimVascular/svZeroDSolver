151template <typename T>
152void Parameter<T>::update(const std::vector<T> &times,
153 const std::vector<T> &values) {
154 this->size = values.size();
155 if (this->size == 1) {
156 this->value = values[0];
157 this->isconstant = true;
158 } else {
159 this->times = times;
160 this->values = values;
161 this->cycle_period = times.back() - times[0];
162 this->isconstant = false;
163 }
164}

169template <typename T>
170T Parameter<T>::get(T time) {
171 // Return the constant value if parameter is constant
172 if (isconstant) {
173 return value;
174 }
175
176 // Determine the time within this->times (necessary to extrapolate)
177 T rtime;
178 if (isperiodic == true) {
179 rtime = fmod(time, cycle_period);
180 } else {
181 // this->times is not periodic when running with external solver
182 rtime = time;
183 }
184
185 // Determine the lower and upper element for interpolation
186 auto i = lower_bound(times.begin(), times.end(), rtime);
187 unsigned int k = i - times.begin();
188 if (i == times.end())
189 --i;
190 else if (*i == rtime) {
191 return values[k];
192 }
193 unsigned int l = k ? k - 1 : 1;
194
195 // Perform linear interpolation
196 // TODO: Implement periodic cubic spline
197 return values[l] +
198 ((values[k] - values[l]) / (times[k] - times[l])) * (rtime - times[l]);
199}

201template <typename T>
202void Parameter<T>::to_steady() {
203 if (isconstant) {
204 return;
205 }
206 value = std::accumulate(values.begin(), values.end(), 0.0) / T(size);
207 isconstant = true;
208 steady_converted = true;
209}
210
211template <typename T>
212void Parameter<T>::to_unsteady() {
213 if (steady_converted) {
214 isconstant = false;
215 steady_converted = false;
216 }
217}
