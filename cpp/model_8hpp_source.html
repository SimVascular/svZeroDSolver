289template <typename T>
290int Model<T>::add_block(BlockType block_type,
291 const std::vector<int> &block_param_ids,
292 std::string_view name, bool internal) {
293 DEBUG_MSG("Adding block " << name << " with type " << block_type);
294 std::shared_ptr<Block<T>> block;
295 switch (block_type) {
296 case BlockType::BLOODVESSEL:
297 block = std::shared_ptr<Block<T>>(
298 new BloodVessel<T>(block_count, block_param_ids, this));
299 break;
300 case BlockType::JUNCTION:
301 block = std::shared_ptr<Block<T>>(
302 new Junction<T>(block_count, block_param_ids, this));
303 break;
304 case BlockType::BLOODVESSELJUNCTION:
305 block = std::shared_ptr<Block<T>>(
306 new BloodVesselJunction<T>(block_count, block_param_ids, this));
307 break;
308 case BlockType::RESISTIVEJUNCTION:
309 block = std::shared_ptr<Block<T>>(
310 new ResistiveJunction<T>(block_count, block_param_ids, this));
311 break;
312 case BlockType::FLOWBC:
313 block = std::shared_ptr<Block<T>>(
314 new FlowReferenceBC<T>(block_count, block_param_ids, this));
315 break;
316 case BlockType::RESISTANCEBC:
317 block = std::shared_ptr<Block<T>>(
318 new ResistanceBC<T>(block_count, block_param_ids, this));
319 break;
320 case BlockType::WINDKESSELBC:
321 block = std::shared_ptr<Block<T>>(
322 new WindkesselBC<T>(block_count, block_param_ids, this));
323 break;
324 case BlockType::PRESSUREBC:
325 block = std::shared_ptr<Block<T>>(
326 new PressureReferenceBC<T>(block_count, block_param_ids, this));
327 break;
328 case BlockType::OPENLOOPCORONARYBC:
329 block = std::shared_ptr<Block<T>>(
330 new OpenLoopCoronaryBC<T>(block_count, block_param_ids, this));
331 break;
332 case BlockType::CLOSEDLOOPCORONARYLEFTBC:
333 block = std::shared_ptr<Block<T>>(
334 new ClosedLoopCoronaryBC<T, MODEL::Side::LEFT>(
335 block_count, block_param_ids, this));
336 break;
337 case BlockType::CLOSEDLOOPCORONARYRIGHTBC:
338 block = std::shared_ptr<Block<T>>(
339 new ClosedLoopCoronaryBC<T, MODEL::Side::RIGHT>(
340 block_count, block_param_ids, this));
341 break;
342 case BlockType::CLOSEDLOOPRCRBC:
343 block = std::shared_ptr<Block<T>>(
344 new ClosedLoopRCRBC<T>(block_count, block_param_ids, this));
345 break;
346 case BlockType::CLOSEDLOOPHEARTPULMONARY:
347 block = std::shared_ptr<Block<T>>(
348 new ClosedLoopHeartPulmonary<T>(block_count, block_param_ids, this));
349 break;
350 default:
351 throw std::runtime_error(
352 "Adding block to model failed: Invalid block type!");
353 }
354 auto name_string = static_cast<std::string>(name);
355 if (internal) {
356 hidden_blocks.push_back(block);
357 } else {
358 blocks.push_back(block);
359 }
360 block_types.push_back(block_type);
361 block_index_map.insert({name_string, block_count});
362 block_names.push_back(name_string);
363 return block_count++;
364}

366template <typename T>
367Block<T> *Model<T>::get_block(std::string_view name) {
368 auto name_string = static_cast<std::string>(name);
369 if (block_index_map.find(name_string) == block_index_map.end()) {
370 return nullptr;

421template <typename T>
422int Model<T>::add_parameter(const std::vector<T> &times,
423 const std::vector<T> &values, bool periodic) {
424 auto param = Parameter<T>(parameter_count, times, values, periodic);
425 if (periodic && (param.isconstant == false)) {
426 if ((this->cardiac_cycle_period > 0.0) &&
427 (param.cycle_period != this->cardiac_cycle_period)) {
428 throw std::runtime_error(
429 "Inconsistent cardiac cycle period defined in parameters");
430 }
431 this->cardiac_cycle_period = param.cycle_period;
432 }
433 parameter_values.push_back(param.get(0.0));
434 parameters.push_back(std::move(param));
435 return parameter_count++;
436}

453template <typename T>
454void Model<T>::finalize() {
455 DEBUG_MSG("Setup degrees-of-freedom of nodes");
456 for (auto &node : nodes) {
457 node->setup_dofs(dofhandler);
458 }
459 DEBUG_MSG("Setup degrees-of-freedom of blocks");
460 for (auto &block : blocks) {
461 block->setup_dofs(dofhandler);
462 }
463 DEBUG_MSG("Setup model-dependent parameters");
464 for (auto &block : blocks) {
465 block->setup_model_dependent_params();
466 }
467 if (cardiac_cycle_period < 0.0) {
468 cardiac_cycle_period = 1.0;
469 }
470}

508template <typename T>
509void Model<T>::to_steady() {
510 for (auto &param : parameters) {
511 param.to_steady();
512 }
513 for (size_t i = 0; i < get_num_blocks(true); i++) {
514 get_block(i)->steady = true;
515 if ((block_types[i] == BlockType::WINDKESSELBC) ||
516 (block_types[i] == BlockType::CLOSEDLOOPRCRBC)) {
517 int param_id_capacitance = blocks[i]->global_param_ids[1];
518 T value = parameters[param_id_capacitance].get(0.0);
519 param_value_cache.insert({param_id_capacitance, value});
520 parameters[param_id_capacitance].update(0.0);
521 }
522 }
523}
524
525template <typename T>
526void Model<T>::to_unsteady() {
527 for (auto &param : parameters) {
528 param.to_unsteady();
529 }
530 for (auto &[param_id_capacitance, value] : param_value_cache) {
531 DEBUG_MSG("Setting Windkessel capacitance back to " << value);
532 parameters[param_id_capacitance].update(value);
533 }
534 for (size_t i = 0; i < get_num_blocks(true); i++) {
535 get_block(i)->steady = false;
536 }
537}

539template <typename T>
540std::map<std::string, int> Model<T>::get_num_triplets() {
541 std::map<std::string, int> num_triplets = {
542 {"F", 0},
543 {"E", 0},
544 {"D", 0},
545 };
546 for (auto &elem : blocks) {
547 for (auto &[key, value] : elem->get_num_triplets()) {
548 num_triplets[key] += value;
549 }
550 }
551 return num_triplets;
552}
