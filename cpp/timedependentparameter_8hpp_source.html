110template <typename T>
111TimeDependentParameter<T>::TimeDependentParameter() {}

113template <typename T>
114TimeDependentParameter<T>::TimeDependentParameter(std::vector<T> times,
115 std::vector<T> values) {
116 this->times = times;
117 this->values = values;
118 size = values.size();
119 if (size == 1) {
120 isconstant = true;
121 cycle_period = 1.0;
122 } else {
123 isconstant = false;
124 cycle_period = times.back() - times[0];
125 }
126}

128template <typename T>
129TimeDependentParameter<T>::~TimeDependentParameter() {}

131template <typename T>
132T TimeDependentParameter<T>::get(T time) {
133 // Return the first and only value if parameter is constant
134 if (isconstant) {
135 return values[0];
136 }
137
138 // Determine the time within a cycle (necessary to extrapolate)
139 T rtime = fmod(time, cycle_period);
140
141 // Determine the lower and upper element for interpolation
142 auto i = lower_bound(times.begin(), times.end(), rtime);
143 unsigned int k = i - times.begin();
144 if (i == times.end())
145 --i;
146 else if (*i == rtime) {
147 return values[k];
148 }
149 unsigned int l = k ? k - 1 : 1;
150
151 // Perform linear interpolation
152 return values[l] +
153 ((values[k] - values[l]) / (times[k] - times[l])) * (rtime - times[l]);
154}

156template <typename T>
157void TimeDependentParameter<T>::to_steady() {
158 T mean =
159 std::accumulate(values.begin(), values.end(), 0.0) / T(values.size());
160 values_cache = values;
161 times_cache = times;
162 size_cache = size;
163 values = std::vector<T>();
164 times = std::vector<T>();
165 values.push_back(mean);
166 values.push_back(0.0);
167 size = 1;
168 isconstant = true;
169}
170
171template <typename T>
172void TimeDependentParameter<T>::to_unsteady() {
173 values = values_cache;
174 times = times_cache;
175 size = size_cache;
176 if (size > 1) {
177 isconstant = false;
178 }
179}
