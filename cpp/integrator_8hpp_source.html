45/**
46 * @brief Generalized-alpha integrator
47 *
48 * This class handles the time integration scheme for solving 0D blood
49 * flow system.
50 *
51 * Flow rate, pressure, and other hemodynamic quantities in 0D models of
52 * vascular anatomies are governed by a system of nonlinear
53 * differential-algebraic equations (DAEs):
54 *
55 * \f[
56 * \mathbf{E}(\mathbf{y}, t) \cdot \dot{\mathbf{y}}+\mathbf{F}(\mathbf{y}, t)
57 * \cdot \mathbf{y}+\mathbf{c}(\mathbf{y}, t)=\mathbf{0} \f]
58 *
59 * Here, \f$y\f$ is the vector of solution quantities and \f$\dot{y}\f$ is the
60 * time derivative of \f$y\f$. \f$N\f$ is the total number of equations and the
61 * total number of global unknowns. The DAE system is solved implicitly using
62 * the generalized-\f$\alpha\f$ method \cite JANSEN2000305.
63 *
64 * @tparam T Scalar type (e.g. `float`, `double`)
65 * `ALGEBRA::SparseSystem`)
66 */
67template <typename T>
68class Integrator {
69 private:
70 T alpha_m;
71 T alpha_f;
72 T alpha_m_inv;
73 T alpha_f_inv;
74 T gamma;
75 T gamma_inv;
76 T time_step_size;
77 T time_step_size_inv;
78 T y_dot_coeff;
79 T atol;
80 T y_init_coeff;
81 T ydot_init_coeff;
82 int max_iter;
83 int size;
84 Eigen::Matrix<T, Eigen::Dynamic, 1> y_af;
85 Eigen::Matrix<T, Eigen::Dynamic, 1> ydot_am;
86 SparseSystem<T> system;
87
88 public:
89 /**
90 * @brief Construct a new Integrator object
91 *
92 * @param system System of equations to integrate
93 * @param time_step_size Time step size for generalized-alpha step
94 * @param rho Spectral radius for generalized-alpha step
95 * @param atol Absolut tolerance for non-linear iteration termination
96 * @param max_iter Maximum number of non-linear iterations
97 */
98 Integrator(MODEL::Model<T> &model, T time_step_size, T rho, T atol,
99 int max_iter);
100
101 /**
102 * @brief Destroy the Integrator object
103 *
104 */
105 ~Integrator();

118template <typename T>
119Integrator<T>::Integrator(MODEL::Model<T> &model, T time_step_size, T rho,
120 T atol, int max_iter) {
121 alpha_m = 0.5 * (3.0 - rho) / (1.0 + rho);
122 alpha_f = 1.0 / (1.0 + rho);
123 alpha_m_inv = 1.0 / alpha_m;
124 alpha_f_inv = 1.0 / alpha_f;
125 gamma = 0.5 + alpha_m - alpha_f;
126 gamma_inv = 1.0 / gamma;
127
128 y_init_coeff = alpha_f * 0.5 * time_step_size;
129 ydot_init_coeff = (1.0 + alpha_m * ((gamma - 0.5) * gamma_inv - 1.0));
130
131 size = model.dofhandler.size();
132 system = SparseSystem<T>(size);
133 this->time_step_size = time_step_size;
134 this->atol = atol;
135 this->max_iter = max_iter;
136 time_step_size_inv = 1.0 / time_step_size;
137
138 y_af = Eigen::Matrix<T, Eigen::Dynamic, 1>(size);
139 ydot_am = Eigen::Matrix<T, Eigen::Dynamic, 1>(size);
140
141 y_dot_coeff = alpha_m * alpha_f_inv * gamma_inv * time_step_size_inv;
142
143 // Make some memory reservations
144 system.reserve(model);
145}

147template <typename T>
148Integrator<T>::~Integrator() {}

150template <typename T>
151State<T> Integrator<T>::step(State<T> &old_state, T time,
152 MODEL::Model<T> &model) {
153 // DEBUG_MSG("Step: time = "<< time);
154 // Predictor + initiator step
155 y_af.setZero();
156 ydot_am.setZero();
157 y_af += old_state.y + old_state.ydot * y_init_coeff;
158 ydot_am += old_state.ydot * ydot_init_coeff;
159
160 // Determine new time
161 T new_time = time + alpha_f * time_step_size;
162
163 // Update time-dependent element contributions in system
164 model.update_time(system, new_time);
165
166 for (size_t i = 0; i < max_iter; i++) {
167 // Update solution-dependent element contribitions
168 model.update_solution(system, y_af);
169
170 // Update residuum and check termination criteria
171 system.update_residual(y_af, ydot_am);
172 if (system.residual.cwiseAbs().maxCoeff() < atol) {
173 break;
174 }
175
176 // Abort if maximum number of non-linear iterations is reached
177 else if (i == max_iter - 1) {
178 throw std::runtime_error(
179 "Maxium number of non-linear iterations reached.");
180 }
181
182 // Determine jacobian
183 system.update_jacobian(y_dot_coeff);
184
185 // Solve system
186 system.solve();
187
188 // Add increment to solution
189 y_af += system.dy;
190 ydot_am += system.dy * y_dot_coeff;
191 }
192
193 // Set new state
194 State<T> new_state = State<T>::Zero(size);
195 new_state.y += old_state.y + (y_af - old_state.y) * alpha_f_inv;
196 new_state.ydot += old_state.ydot + (ydot_am - old_state.ydot) * alpha_m_inv;
197
198 return new_state;
199}
