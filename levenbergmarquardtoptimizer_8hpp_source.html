44/**
45 * @brief Levenberg-Marquardt optimization class
46 *
47 * The 0D residual (assuming no time-dependency in parameters) is
48 *
49 * \f[
50 * \boldsymbol{r}(\boldsymbol{\alpha}, \boldsymbol{y}, \boldsymbol{\dot{y}}) =
51 * \boldsymbol{E}(\boldsymbol{\alpha}, \boldsymbol{y}) \cdot
52 * \dot{\boldsymbol{y}}+\boldsymbol{F}(\boldsymbol{\alpha}, \boldsymbol{y})
53 * \cdot \boldsymbol{y}+\boldsymbol{c}(\boldsymbol{\alpha}, \boldsymbol{y}) \f]
54 *
55 * with solution vector \f$\boldsymbol{y} \in \mathbb{R}^{N}\f$ (flow and
56 * pressure at nodes), LPN parameters \f$\boldsymbol{\alpha} \in
57 * \mathbb{R}^{P}\f$, system matrices \f$\boldsymbol{E},\boldsymbol{F} \in
58 * \mathbb{R}^{NxN}\f$, and system vector \f$\boldsymbol{c} \in
59 * \mathbb{R}^{N}\f$.
60 *
61 * The least squares problem can be formulated as
62 *
63 * \f[
64 * \min _\alpha S, \quad \mathrm { with } \quad S=\sum_i^D
65 * r_i^2\left(\boldsymbol{\alpha}, y_i, \dot{y}_i\right) \f]
66 *
67 * with given solution vectors \f$\boldsymbol{y}\f$, \f$\boldsymbol{\dot{y}}\f$
68 * at all datapoints \f$D\f$. The parameter vector is iteratively improved
69 * according to
70 *
71 * \f[
72 * \boldsymbol{\alpha}^{i+1}=\boldsymbol{\alpha}^{i}+\Delta
73 * \boldsymbol{\alpha}^{i+1} \f]
74 *
75 * wherein the increment \f$\Delta \boldsymbol{\alpha}^{i+1} \f$ is determined
76 * by solving the following system:
77 *
78 * \f[
79 * \left[\mathbf{J}^{\mathrm{T}} \mathbf{J}+\lambda
80 * \operatorname{diag}\left(\mathbf{J}^{\mathrm{T}} \mathbf{J}\right)\right]^{i}
81 * \cdot \Delta \boldsymbol{\alpha}^{i+1}=-\left[\mathbf{J}^{\mathrm{T}}
82 * \mathbf{r}\right]^{i}, \quad \lambda^{i}=\lambda^{i-1}
83 * \cdot\left\|\left[\mathbf{J}^{\mathrm{T}} \mathbf{r}\right]^{i}\right\|_2
84 * /\left\|\left[\mathbf{J}^{\mathrm{T}} \mathbf{r}\right]^{i-1}\right\|_2. \f]
85 *
86 * The algorithm terminates when the following tolerance thresholds are reached
87 *
88 * \f[
89 * \left\|\left[\mathbf{J}^{\mathrm{T}}
90 * \mathbf{r}\right]^{\mathrm{i}}\right\|_2<\operatorname{tol}_{\text {grad
91 * }}^\alpha \text { and }\left\|\Delta
92 * \boldsymbol{\alpha}^{\mathrm{i}+1}\right\|_2<\mathrm{tol}_{\text {inc
93 * }}^\alpha, \f]
94 *
95 * The Jacobian is derived from the residual as
96 *
97 * \f[
98 * J = \frac{\partial \boldsymbol{r}}{\partial \boldsymbol{\alpha}} =
99 * \frac{\partial \mathbf{E}}{\partial \boldsymbol{\alpha}} \cdot
100 * \dot{\mathbf{y}}+\frac{\partial \mathbf{F}}{\partial \boldsymbol{\alpha}}
101 * \cdot \mathbf{y}+\frac{\partial \mathbf{c}}{\partial \boldsymbol{\alpha}} \f]
102 *
103 *
104 * @tparam T Scalar type (e.g. `float`, `double`)
105 */
106template <typename T>
107class LevenbergMarquardtOptimizer {
108 public:
109 /**
110 * @brief Construct a new LevenbergMarquardtOptimizer object
111 *
112 * @param model The 0D model
113 * @param num_obs Number of observations in optimization
114 * @param num_params Number of parameters in optimization
115 * @param lambda0 Initial damping factor
116 * @param tol_grad Gradient tolerance
117 * @param tol_inc Parameter increment tolerance
118 * @param max_iter Maximum iterations
119 */
120 LevenbergMarquardtOptimizer(MODEL::Model<T>* model, int num_obs,
121 int num_params, T lambda0, T tol_grad, T tol_inc,
122 int max_iter);
123
124 /**
125 * @brief Destroy the LevenbergMarquardtOptimizer object
126 *
127 */
128 ~LevenbergMarquardtOptimizer();
129
130 /**
131 * @brief Run the optimization algorithm
132 *
133 * @param alpha Initial parameter vector alpha
134 * @param y_obs Matrix (num_obs x n) with all observations for y
135 * @param dy_obs Matrix (num_obs x n) with all observations for dy
136 * @return Eigen::Matrix<T, Eigen::Dynamic, 1> Optimized parameter vector
137 * alpha
138 */
139 Eigen::Matrix<T, Eigen::Dynamic, 1> run(
140 Eigen::Matrix<T, Eigen::Dynamic, 1> alpha,
141 std::vector<std::vector<T>>& y_obs, std::vector<std::vector<T>>& dy_obs);
142
143 private:
144 Eigen::SparseMatrix<T> jacobian;
145 Eigen::Matrix<T, Eigen::Dynamic, 1> residual;
146 Eigen::Matrix<T, Eigen::Dynamic, 1> delta;
147 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> mat;
148 Eigen::Matrix<T, Eigen::Dynamic, 1> vec;
149 MODEL::Model<T>* model;
150 T lambda;
151
152 int num_obs;
153 int num_params;
154 int num_eqns;
155 int num_vars;
156 int num_dpoints;
157
158 T tol_grad;
159 T tol_inc;
160 int max_iter;
161
162 void update_gradient(Eigen::Matrix<T, Eigen::Dynamic, 1>& alpha,
163 std::vector<std::vector<T>>& y_obs,
164 std::vector<std::vector<T>>& dy_obs);
165
166 void update_delta(bool first_step);
167};

169template <typename T>
170LevenbergMarquardtOptimizer<T>::LevenbergMarquardtOptimizer(
171 MODEL::Model<T>* model, int num_obs, int num_params, T lambda0, T tol_grad,
172 T tol_inc, int max_iter) {
173 this->model = model;
174 this->num_obs = num_obs;
175 this->num_params = num_params;
176 this->num_eqns = model->dofhandler.get_num_equations();
177 this->num_vars = model->dofhandler.get_num_variables();
178 this->num_dpoints = this->num_obs * this->num_eqns;
179 this->lambda = lambda0;
180 this->tol_grad = tol_grad;
181 this->tol_inc = tol_inc;
182 this->max_iter = max_iter;
183 jacobian = Eigen::SparseMatrix<T>(num_dpoints, num_params);
184 residual = Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(num_dpoints);
185 mat =
186 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>(num_params, num_params);
187 vec = Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(num_params);
188}

193template <typename T>
194Eigen::Matrix<T, Eigen::Dynamic, 1> LevenbergMarquardtOptimizer<T>::run(
195 Eigen::Matrix<T, Eigen::Dynamic, 1> alpha,
196 std::vector<std::vector<T>>& y_obs, std::vector<std::vector<T>>& dy_obs) {
197 for (size_t i = 0; i < max_iter; i++) {
198 update_gradient(alpha, y_obs, dy_obs);
199 if (i == 0) {
200 update_delta(true);
201 } else {
202 update_delta(false);
203 }
204 alpha -= delta;
205 T norm_grad = vec.norm();
206 T norm_inc = delta.norm();
207 std::cout << std::setprecision(1) << std::scientific << "Iteration "
208 << i + 1 << " | lambda: " << lambda << " | norm inc: " << norm_inc
209 << " | norm grad: " << norm_grad << std::endl;
210 if ((norm_grad < tol_grad) && (norm_inc < tol_inc)) {
211 break;
212 }
213 if (i >= max_iter - 1) {
214 std::cout << "Maximum number of iterations reached" << std::endl;
215 break;
216 }
217 }
218 return alpha;
219}

221template <typename T>
222void LevenbergMarquardtOptimizer<T>::update_gradient(
223 Eigen::Matrix<T, Eigen::Dynamic, 1>& alpha,
224 std::vector<std::vector<T>>& y_obs, std::vector<std::vector<T>>& dy_obs) {
225 // Set jacobian and residual to zero
226 jacobian.setZero();
227 residual.setZero();
228
229 // Assemble gradient and residual
230 for (size_t i = 0; i < num_obs; i++) {
231 for (size_t j = 0; j < model->get_num_blocks(true); j++) {
232 auto block = model->get_block(j);
233 for (size_t l = 0; l < block->global_eqn_ids.size(); l++) {
234 block->global_eqn_ids[l] += num_eqns * i;
235 }
236 block->update_gradient(jacobian, residual, alpha, y_obs[i], dy_obs[i]);
237 for (size_t l = 0; l < block->global_eqn_ids.size(); l++) {
238 block->global_eqn_ids[l] -= num_eqns * i;
239 }
240 }
241 }
242}

244template <typename T>
245void LevenbergMarquardtOptimizer<T>::update_delta(bool first_step) {
246 // Cache old gradient vector and calulcate new one
247 Eigen::Matrix<T, Eigen::Dynamic, 1> vec_old = vec;
248 vec = jacobian.transpose() * residual;
249
250 // Determine new lambda parameter from new and old gradient vector
251 if (!first_step) {
252 lambda *= vec.norm() / vec_old.norm();
253 }
254
255 // Determine gradient matrix
256 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> jacobian_sq =
257 jacobian.transpose() * jacobian;
258 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> jacobian_sq_diag =
259 jacobian_sq.diagonal().asDiagonal();
260 mat = jacobian_sq + lambda * jacobian_sq_diag;
261
262 // Solve for new delta
263 delta = mat.llt().solve(vec);
264}
