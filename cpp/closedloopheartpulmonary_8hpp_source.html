16 /**
17 * @brief RCR boundary condition element
18 *
19 * Models the mechanical behavior of a Windkessel boundary condition.
20 *
21 * \f[
22 * \begin{circuitikz} \draw
23 * node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
24 * \draw (1,0) node[anchor=south]{$P_{in}$}
25 * to [R, l=$R_p$, *-] (3,0)
26 * to [R, l=$R_d$, *-*] (5,0)
27 * node[anchor=south]{$P_{ref}$}
28 * (3,0) to [C, l=$C$, *-] (3,-1.5)
29 * node[ground]{$P_{C}$};
30 * \end{circuitikz}
31 * \f]
32 *
33 * ### Governing equations
34 *
35 * \f[
36 * R_{d} Q^{e}-P_{c}^{e}+P_{r e f}-R_{d} C \frac{d P_{c}^{e}}{d t}=0
37 * \f]
38 *
39 * \f[
40 * P^{e}-P_{c}^{e}-R_{p} Q^{e}=0
41 * \f]
42 *
43 * ### Local contributions
44 *
45 * \f[
46 * \mathbf{y}^{e}=\left[\begin{array}{lll}P^{e} & Q^{e} & P_{c}^{e}\end{array}\right]^{T}
47 * \f]
48 *
49 * \f[
50 * \mathbf{E}^{e}=\left[\begin{array}{ccc}
51 * 0 & 0 & -R_{d} C \\
52 * 0 & 0 & 0
53 * \end{array}\right]
54 * \f]
55 *
56 * \f[
57 * \mathbf{F}^{e}=\left[\begin{array}{ccc}
58 * 0 & R_{d} & -1 \\
59 * 1 & -R_{p} & -1
60 * \end{array}\right]
61 * \f]
62 *
63 * \f[
64 * \mathbf{c}^{e}=\left[\begin{array}{c}
65 * P_{r e f} \\
66 * 0
67 * \end{array}\right]
68 * \f]
69 *
70 * @tparam T Scalar type (e.g. `float`, `double`)
71 */

307 void ClosedLoopHeartPulmonary<T>::update_solution(ALGEBRA::SparseSystem<T> &system, Eigen::Matrix<T, Eigen::Dynamic, 1> &y)
308 {
309
310 T psi_ra, psi_la, psi_ra_derivative, psi_la_derivative;
311 this->get_psi_ra_la(y, &psi_ra, &psi_la, &psi_ra_derivative, &psi_la_derivative)
312
313 T valves[16]
314 this->get_valve_positions(y, &valves)
315
316 // F and C matrices depend on time and solution
317 // Specifying all terms here, including constant terms (which can instead be specified in init) for readability
318 // (since F and C will need to be reassambleed anyway)
319 // DOF IDs are arranged as inflow [P_in,Q_in,P_out,Q_out,internal variables...] (see assign_global_ids() in connections.py)
320
321 // DOF 0, Eq 0: Right atrium pressure
322 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[0]) = 1.0
323 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[4]) = -this->AA*params.Emax_ra
324 system.C(this->global_eqn_ids[0]) = this->AA*params.Emax_ra*params.Vaso_ra + psi_ra*(this->AA-1.0)
325 system.D.coeffRef(this->global_eqn_ids[0], this->global_var_ids[4]) = psi_ra_derivative*(this->AA-1.0)
326
327 // DOF 1: Flow into right atrium (no equation)
328
329 // DOF 2, Eq 1: Aortic pressure
330 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[15]) = -valves[15]
331 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[3]) = 1.0
332
333 // DOF 3: Flow into aorta (no equation)
334
335 // DOF 4, Eq 2: Right atrium volume
336 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[5]) = 1.0*valves[5]
337 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[1]) = -1.0
338
339 // DOF 5, Eq 3: Right atrium outflow
340 system.F.coeffRef(this->global_eqn_ids[3], this->global_var_ids[5]) = params.Rra_v*valves[5]
341 system.F.coeffRef(this->global_eqn_ids[3], this->global_var_ids[0]) = -1.0
342 system.F.coeffRef(this->global_eqn_ids[3], this->global_var_ids[6]) = 1.0
343
344 // DOF 6, Eq 4: Right ventricle pressure
345 system.F.coeffRef(this->global_eqn_ids[4], this->global_var_ids[6]) = 1.0
346 system.F.coeffRef(this->global_eqn_ids[4], this->global_var_ids[7]) = -this->Erv
347 system.C(this->global_eqn_ids[4]) = this->Erv*params.Vrv_u
348
349 // DOF 7, Eq 5: Right ventricle volume
350 system.F.coeffRef(this->global_eqn_ids[5], this->global_var_ids[5]) = -1.0*valves[5]
351 system.F.coeffRef(this->global_eqn_ids[5], this->global_var_ids[8]) = 1.0*valves[8]
352
353 // DOF 8, Eq 6: Right ventricle outflow
354 system.F.coeffRef(this->global_eqn_ids[6], this->global_var_ids[6]) = -1.0
355 system.F.coeffRef(this->global_eqn_ids[6], this->global_var_ids[9]) = 1.0
356 system.F.coeffRef(this->global_eqn_ids[6], this->global_var_ids[8]) = params.Rrv_a*valves[8]
357
358 // DOF 9, Eq 7: Pulmonary pressure
359 system.F.coeffRef(this->global_eqn_ids[7], this->global_var_ids[8]) = -valves[8]
360 system.F.coeffRef(this->global_eqn_ids[7], this->global_var_ids[9]) = 1.0/params.Rpd
361 system.F.coeffRef(this->global_eqn_ids[7], this->global_var_ids[10]) = -1.0/params.Rpd
362
363 // DOF 10, Eq 8: Left atrium pressure
364 system.F.coeffRef(this->global_eqn_ids[8], this->global_var_ids[10]) = 1.0
365 system.F.coeffRef(this->global_eqn_ids[8], this->global_var_ids[11]) = -this->AA*params.Emax_la
366 system.C(this->global_eqn_ids[8]) = this->AA*params.Emax_la*params.Vaso_la + psi_la*(this->AA-1.0)
367 system.D.coeffRef(this->global_eqn_ids[8], this->global_var_ids[11]) = psi_la_derivative*(this->AA-1.0)
368
369 // DOF 11, Eq 9: Left atrium volume
370 system.F.coeffRef(this->global_eqn_ids[9], this->global_var_ids[8]) = -1.0*valves[8]
371 system.F.coeffRef(this->global_eqn_ids[9], this->global_var_ids[12]) = 1.0*valves[12]
372
373 // DOF 12, Eq 10: Left atrium outflow
374 system.F.coeffRef(this->global_eqn_ids[10], this->global_var_ids[10]) = -1.0
375 system.F.coeffRef(this->global_eqn_ids[10], this->global_var_ids[13]) = 1.0
376 system.F.coeffRef(this->global_eqn_ids[10], this->global_var_ids[12]) = params.Rla_v*valves[12]
377
378 // DOF 13, Eq 11: Left ventricle pressure
379 system.F.coeffRef(this->global_eqn_ids[11], this->global_var_ids[13]) = 1.0
380 system.F.coeffRef(this->global_eqn_ids[11], this->global_var_ids[14]) = -this->Elv
381 system.C(this->global_eqn_ids[11]) = this->Elv*params.Vlv_u
382
383 // DOF 14, Eq 12: Left ventricle volume
384 system.F.coeffRef(this->global_eqn_ids[12], this->global_var_ids[12]) = -1.0*valves[12]
385 system.F.coeffRef(this->global_eqn_ids[12], this->global_var_ids[15]) = 1.0*valves[15]
386
387 // DOF 15, Eq 13: Left ventricle outflow
388 system.F.coeffRef(this->global_eqn_ids[13], this->global_var_ids[13]) = -1.0
389 system.F.coeffRef(this->global_eqn_ids[13], this->global_var_ids[2]) = 1.0
390 system.F.coeffRef(this->global_eqn_ids[13], this->global_var_ids[15]) = params.Rlv_ao*valves[15]

394 template <typename T>
395 void ClosedLoopHeartPulmonary<T>::get_activation_and_elastance_functions(T time)
396 {
397 T T_cardiac = this->cardiac_cycle_perdiod;
398 T Tsa = T_cardiac*params.Tsa;
399 T tpwave = T_cardiac/params.tpwave;
400 T t_in_cycle = time%T_cardiac;
401
402 // Activation function
403 T AA = 0.0;
404 if (t_in_cycle <= tpwave)
405 {
406 AA = (0.5)*(1.0-cos(2.0*PI*(t_in_cycle-tpwave+Tsa)/Tsa));
407 }
408 else if ((t_in_cycle >= (T_cardiac-Tsa)+tpwave) and (t_in_cycle < T_cardiac))
409 {
410 AA = (0.5)*(1.0-cos(2.0*PI*(t_in_cycle-tpwave-(T_cardiac-Tsa))/Tsa));
411 }
412 else
413 {
414 AA = 0.0;
415 }
416
417 // Elastance modes (copied from J. Tran's tuning framework)
418 int num_elast_modes = 25;
419 T Ft_elastance[num_elast_modes][2];
420 T Ft_elastance = {{ 0.283748803 , 0.000000000},
421 { 0.031830626 , -0.374299825},
422 {-0.209472400 , -0.018127770},
423 { 0.020520047 , 0.073971113},
424 { 0.008316883 , -0.047249597},
425 {-0.041677660 , 0.003212163},
426 { 0.000867323 , 0.019441411},
427 {-0.001675379 , -0.005565534},
428 {-0.011252277 , 0.003401432},
429 {-0.000414677 , 0.008376795},
430 { 0.000253749 , -0.000071880},
431 {-0.002584966 , 0.001566861},
432 { 0.000584752 , 0.003143555},
433 { 0.000028502 , -0.000024787},
434 { 0.000022961 , -0.000007476},
435 { 0.000018735 , -0.000001281},
436 { 0.000015573 , 0.000001781},
437 { 0.000013133 , 0.000003494},
438 { 0.000011199 , 0.000004507},
439 { 0.000009634 , 0.000005117},
440 { 0.000008343 , 0.000005481},
441 { 0.000007265 , 0.000005687},
442 { 0.000006354 , 0.000005789},
443 { 0.000005575 , 0.000005821},
444 { 0.000004903 , 0.000005805}};
445
446 # RV and LV elastance
447 T Elv_i = 0.0;
448 for(auto i=0; i < num_elast_modes; i++):
449 Elv_i = Elv_i + (Ft_elastance[i][0])*cos(2.0*PI*i*t_in_cycle/T_cardiac)
450 - (Ft_elastance[i][1])*sin(2.0*PI*i*t_in_cycle/T_cardiac);
451
452 this->Elv = Elv_i*params.Elv_s;
453 this->Erv = Elv_i*params.Erv_s;
454
455 }

457 template <typename T>
458 void ClosedLoopHeartPulmonary<T>::get_psi_ra_la(Eigen::Matrix<T, Eigen::Dynamic, 1> &y, T* psi_ra, T* psi_la, T* psi_ra_derivative, T* psi_la_derivative)
459 {
460 auto RA_volume = y[this->global_var_ids[4]];
461 auto LA_volume = y[this->global_var_ids[11]];
462 psi_ra = params.Kxp_ra*(exp((RA_volume-params.Vaso_ra)*params.Kxv_ra)-1.0);
463 psi_la = params.Kxp_la*(exp((LA_volume-params.Vaso_la)*params.Kxv_la)-1.0);
464
465 psi_ra_derivative = params.Kxp_ra*exp((RA_volume-params.Vaso_ra)*params.Kxv_ra)*params.Kxv_ra;
466 psi_la_derivative = params.Kxp_la*exp((LA_volume-params.Vaso_la)*params.Kxv_la)*params.Kxv_la;
467

470 template <typename T>
471 void ClosedLoopHeartPulmonary<T>::get_valve_positions(Eigen::Matrix<T, Eigen::Dynamic, 1> &y, T* valves)
472 {
473 std::fill(valves, valves + 16, 1.0);
474
475 // RA to RV
476 //auto pressure_ra = y[this->inlet_nodes[0]->pres_dof];
477 auto pressure_ra = y[this->global_var_ids[0]];
478 auto pressure_rv = y[this->global_var_ids[6]];
479 auto outflow_ra = y[this->global_var_ids[5]];
480 if ((pressure_ra <= pressure_rv) and (outflow_ra <= 0.0))
481 {
482 valves[5] = 0.0;
483 y[this->global_var_ids[5]] = 0.0;
484 }
485
486 // RV to pulmonary
487 auto pressure_pulmonary = y[this->global_var_ids[9]];
488 auto outflow_rv = y[this->global_var_ids[8]];
489 if ((pressure_rv <= pressure_pulmonary) and (outflow_rv <= 0.0))
490 {
491 valves[8] = 0.0;
492 y[this->global_var_ids[8]] = 0.0;
493 }
494
495 // LA to LV
496 auto pressure_la = y[this->global_var_ids[10]];
497 auto pressure_lv = y[this->global_var_ids[13]];
498 auto outflow_la = y[this->global_var_ids[12]];
499 if ((pressure_la <= pressure_lv) and (outflow_la <= 0.0))
500 {
501 valves[12] = 0.0;
502 y[this->global_var_ids[12]] = 0.0;
503 }
504
505 // LV to aorta
506 auto pressure_aorta = y[this->global_var_ids[2]];
507 auto outflow_lv = y[this->global_var_ids[15]];
508 if ((pressure_lv <= pressure_aorta) and (outflow_lv <= 0.0))
509 {
510 valves[15] = 0.0;
511 y[this->global_var_ids[11]] = 0.0;
512 }
513 }
