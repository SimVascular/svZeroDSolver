44/**
45 * @brief Resistor-capacitor-inductor blood vessel with optional stenosis
46 *
47 * Models the mechanical behavior of a bloodvessel with optional stenosis.
48 *
49 * \f[
50 * \begin{circuitikz} \draw
51 * node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
52 * \draw (1,0) node[anchor=south]{$P_{in}$}
53 * to [R, l=$R$, *-] (3,0)
54 * to [R, l=$R_{ste}$, -] (5,0) node[anchor=south] {$P_{C}$}
55 * (5,0) to [L, l=$L$, *-*] (7,0)
56 * node[anchor=south]{$P_{out}$}
57 * (5,0) to [C, l=$C$, *-] (5,-1.5)
58 * node[ground]{};
59 * \draw [-latex] (7.2,0) -- (8,0) node[right] {$Q_{out}$};
60 * \end{circuitikz}
61 * \f]
62 *
63 * ### Governing equations
64 *
65 * \f[
66 * P_{in}^{e}-P_{out}^{e}-(R+R_{ste}) Q_{in}^{e}-L\frac{d Q_{out}^{e}}{dt}=0
67 * \f]
68 *
69 * \f[
70 * Q_{i n}^{e}-Q_{o u t}^{e}-C \frac{d P_{c}^{e}}{d t}=0
71 * \f]
72 *
73 * \f[
74 * P_{i n}^{e}-(R+R_{ste}) Q_{i n}^{e}-P_{c}=0
75 * \f]
76 *
77 * ### Local contributions
78 *
79 * \f[
80 * \mathbf{y}^{e}=\left[\begin{array}{lllll}P_{i n}^{e} & Q_{in}^{e} &
81 * P_{out}^{e} & Q_{out}^{e} & P_C\end{array}\right]^{T} \f]
82 *
83 * \f[
84 * \mathbf{E}^{e}=\left[\begin{array}{ccccc}
85 * 0 & 0 & 0 & -L & 0 \\
86 * 0 & 0 & 0 & 0 & -C \\
87 * 0 & 0 & 0 & 0 & 0
88 * \end{array}\right]
89 * \f]
90 *
91 * \f[
92 * \mathbf{F}^{e}=\left[\begin{array}{ccccc}
93 * 1 & -R_{ste}-R & -1 & 0 & 0 \\
94 * 0 & 1 & 0 & -1 & 0 \\
95 * 1 & -R_{ste}-R & 0 & 0 & -1
96 * \end{array}\right]
97 * \f]
98 *
99 * \f[
100 * \mathbf{D}^{e}=\left[\begin{array}{ccccc}
101 * 0 & -R_{ste} & 0 & 0 & 0 \\
102 * 0 & 0 & 0 & 0 & 0 \\
103 * 0 & -R_{ste} & 0 & 0 & 0
104 * \end{array}\right]
105 * \f]
106 *
107 * with the stenosis resistance \f$ R_{ste}=K_{t} \frac{\rho}{2
108 * A_{o}^{2}}\left(\frac{A_{o}}{A_{s}}-1\right)^{2}|Q_{in}^{e}| \f$. The
109 * constant part of the equation is summarized in \ref
110 * Parameters::stenosis_coefficient. \f$R\f$, \f$C\f$, and \f$L\f$ refer to
111 * Poisieuille resistance, capacitance and inductance, respectively.
112 *
113 * @tparam T Scalar type (e.g. `float`, `double`)
114 */

242template <typename T>
243void BloodVessel<T>::get_block_params(std::vector<T> &block_params) {
244 if (block_params.size() != 4) {
245 throw std::runtime_error(
246 "Wrong vector size in get_block_params for BloodVessel. Size should be "
247 "4 but is currently " +
248 std::to_string(block_params.size()));
249 }
250 block_params[0] = this->params.R;
251 block_params[1] = this->params.C;
252 block_params[2] = this->params.L;
253 block_params[3] = this->params.stenosis_coefficient;
254}

256template <typename T>
257void BloodVessel<T>::update_constant(ALGEBRA::SparseSystem<T> &system) {
258 system.E.coeffRef(this->global_eqn_ids[0], this->global_var_ids[3]) =
259 -params.L;
260 system.E.coeffRef(this->global_eqn_ids[1], this->global_var_ids[4]) =
261 -params.C;
262
263 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[0]) = 1.0;
264 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[1]) =
265 -params.R;
266 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[2]) = -1.0;
267
268 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[1]) = 1.0;
269 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[3]) = -1.0;
270
271 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[0]) = 1.0;
272 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[1]) =
273 -params.R;
274 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[4]) = -1.0;
275}

277template <typename T>
278void BloodVessel<T>::update_solution(ALGEBRA::SparseSystem<T> &system,
279 Eigen::Matrix<T, Eigen::Dynamic, 1> &y) {
280 T q_in = fabs(y[this->global_var_ids[1]]);
281 T fac1 = -params.stenosis_coefficient * q_in;
282 T fac2 = fac1 - params.R;
283 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[1]) = fac2;
284 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[1]) = fac2;
285 system.D.coeffRef(this->global_eqn_ids[0], this->global_var_ids[1]) = fac1;
286 system.D.coeffRef(this->global_eqn_ids[2], this->global_var_ids[1]) = fac1;
287}
