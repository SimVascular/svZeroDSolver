43/**
44 * @brief Open loop coronary boundary condition based on \cite kim_coronary.
45 *
46 * \f[
47 * \begin{circuitikz} \draw
48 * node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
49 * \draw (1,0) node[anchor=south]{$P_{in}$}
50 * to [R, l=$R_a$, *-] (3,0)
51 * to [R, l=$R_{am}$, -] (5,0)
52 * to [R, l=$R_v$, *-*] (7,0)
53 * node[anchor=south]{$P_{v}$}
54 * (5,0) to [C, l=$C_{im} \;V_{im}$, -*] (5,-1.5)
55 * node[left]{$P_{im}$}
56 * (3,0) to [C, l=$C_a$, -*] (3,-1.5)
57 * node[left]{$P_a$};
58 * \end{circuitikz}
59 * \f]
60 *
61 * ### Governing equations
62 *
63 * \f[
64 * C_{i m} R_{v} Q^{e}-V_{i m}^{e}-C_{i m} P_{i m}+C_{i m} P_{v}-C_{i m} R_{v}
65 * \frac{d V_{i m}^{e}}{d t}-C_{a} C_{i m} R_{v} \frac{d P^{e}}{d t}+R_{a} C_{a}
66 * C_{i m} R_{v} \frac{d Q^{e}}{d t}+C_{a} C_{i m} R_{v} \frac{d P_{a}^{e}}{d
67 * t}=0 \f]
68 *
69 * \f[
70 * C_{i m} R_v P^{e}-C_{i m} R_{v} R_{a} Q^{e}-R_{v} V_{i m}^{e}-C_{i m} R_{v}
71 * P_{i m}-C_{i m} R_{v} R_{a m} \frac{d V_{i m}^{e}}{d t}-R_{a m} V_{i
72 * m}^{e}-C_{i m} R_{a m} P_{i m}+R_{a m} C_{i m} P_{v}=0 \f]
73 *
74 * ### Local contributions
75 *
76 * \f[
77 * \mathbf{y}^{e}=\left[\begin{array}{lll}P^{e} & Q^{e} & V_{i
78 * m}^{e}\end{array}\right]^{T}, \f]
79 *
80 * \f[
81 * \mathbf{E}^{e}=\left[\begin{array}{ccc}-C_{a} C_{i m} R_{v} & R_{a} C_{a}
82 * C_{i m} R_{v} & -C_{i m} R_{v} \\ 0 & 0 & -C_{i m} R_{v} R_{a
83 * m}\end{array}\right] \f]
84 *
85 * \f[
86 * \mathbf{F}^{e}=\left[\begin{array}{ccc}0 & C_{i m} R_{v} & -1 \\C_{i m} R_{v}
87 * & -C_{i m} R_{v} R_{a} & -\left(R_{v}+R_{a m}\right)\end{array}\right] \f]
88 *
89 * \f[
90 * \mathbf{c}^{e}=\left[\begin{array}{c}C_{i m}\left(-P_{i m}+P_{v}\right)+C_{a}
91 * C_{i m} R_{v} \frac{d P_{a}}{d t} \\-C_{i m}\left(R_{v}+R_{a m}\right) P_{i
92 * m}+R_{a m} C_{i m} P_{v}\end{array}\right] \f]
93 *
94 * Assume \f$P_a=0\f$.
95 *
96 * @tparam T Scalar type (e.g. `float`, `double`)
97 */

224template <typename T>
225void OpenLoopCoronaryBC<T>::update_constant(ALGEBRA::SparseSystem<T> &system) {
226 if (issteady) {
227 // Different assmembly for steady block to avoid singular system
228 // and solve for the internal variable V_im inherently
229 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[0]) =
230 -params.Cim;
231 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[1]) =
232 params.Cim * (params.Ra + params.Ram);
233 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[2]) = 1.0;
234 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[0]) = -1.0;
235 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[1]) =
236 params.Ra + params.Ram + params.Rv;
237 } else {
238 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[1]) =
239 params.Cim * params.Rv;
240 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[2]) = -1.0;
241 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[0]) =
242 params.Cim * params.Rv;
243 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[1]) =
244 -params.Cim * params.Rv * params.Ra;
245 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[2]) =
246 -(params.Rv + params.Ram);
247
248 system.E.coeffRef(this->global_eqn_ids[0], this->global_var_ids[0]) =
249 -params.Ca * params.Cim * params.Rv;
250 system.E.coeffRef(this->global_eqn_ids[0], this->global_var_ids[1]) =
251 params.Ra * params.Ca * params.Cim * params.Rv;
252 system.E.coeffRef(this->global_eqn_ids[0], this->global_var_ids[2]) =
253 -params.Cim * params.Rv;
254 system.E.coeffRef(this->global_eqn_ids[1], this->global_var_ids[2]) =
255 -params.Cim * params.Rv * params.Ram;
256 }
257}

259template <typename T>
260void OpenLoopCoronaryBC<T>::update_time(ALGEBRA::SparseSystem<T> &system,
261 T time) {
262 T Pim = params.Pim.get(time);
263 T Pv = params.Pv.get(time);
264
265 if (issteady) {
266 system.C(this->global_eqn_ids[0]) = -params.Cim * Pim;
267 system.C(this->global_eqn_ids[1]) = Pv;
268 } else {
269 system.C(this->global_eqn_ids[0]) = params.Cim * (-Pim + Pv);
270 system.C(this->global_eqn_ids[1]) =
271 -params.Cim * (params.Rv + params.Ram) * Pim +
272 params.Ram * params.Cim * Pv;
273 }
274}
