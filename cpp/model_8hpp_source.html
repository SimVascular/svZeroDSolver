166template <typename T>
167std::map<std::string, int> Model<T>::get_num_triplets() {
168 std::map<std::string, int> num_triplets = {
169 {"F", 0},
170 {"E", 0},
171 {"D", 0},
172 };
173 for (auto &elem : blocks) {
174 for (auto &[key, value] : elem->get_num_triplets()) {
175 num_triplets[key] += value;
176 }
177 }
178 return num_triplets;
179}
