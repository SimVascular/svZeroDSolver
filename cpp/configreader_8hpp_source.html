67template <typename T>
68class ConfigReader {
69 public:

73 ConfigReader();
74
75 /**
76 * @brief Destroy the Config Reader object

78 ~ConfigReader();

80 /**
81 * @brief Create model from configuration file
82 *
83 * @return Model
84 */
85 void load(std::string &specifier);
86
87 MODEL::Model<T> *model; ///< Simulation model
88 ALGEBRA::State<T> initial_state; ///< Initial state
89
90 T sim_cardiac_cycle_period; ///< Cardiac cycle period.
91 T sim_time_step_size; ///< Simulation time step size
92 T sim_abs_tol; ///< Absolute tolerance for simulation
93
94 int sim_num_cycles; ///< Number of cardiac cycles to simulate
95 int sim_pts_per_cycle; ///< Number of time steps per cardiac cycle
96 int sim_num_time_steps; ///< Total number of time steps
97 int sim_nliter; ///< Maximum number of non-linear iterations in time
98 ///< integration
99 int output_interval; ///< Interval of writing output
100
101 bool sim_steady_initial; ///< Start from steady solution
102 bool
103 output_variable_based; ///< Output variable based instead of vessel based
104 bool output_mean_only; ///< Output only the mean value
105 bool output_derivative; ///< Output derivatives
106 bool output_last_cycle_only; ///< Output only the last cardiac cycle
107
108 bool sim_coupled; ///< Running 0D simulation coupled with external solver
109 T sim_external_step_size; ///< Step size of external solver if running
110 ///< coupled
111};

119template <typename T>
120void ConfigReader<T>::load(std::string &specifier) {
121 DEBUG_MSG("Start ConfigReader::load");
122 // Initialize model pointer
123 model = new MODEL::Model<T>();
124
125 // Create iterator for json configuration
126 simdjson::ondemand::parser parser;
127 simdjson::padded_string string;
128 if (HELPERS::startswith(specifier, "{")) {
129 string = simdjson::padded_string(specifier);
130 } else {
131 string = simdjson::padded_string::load(specifier);
132 }
133 DEBUG_MSG("Loaded svZeroD config file. Starting parsing.");
134 auto config = parser.iterate(string);
135
136 // Load simulation parameters
137 auto sim_params = config["simulation_parameters"];
138 try {
139 sim_coupled = sim_params["coupled_simulation"].get_bool();
140 } catch (simdjson::simdjson_error) {
141 sim_coupled = false;
142 }
143 if (!sim_coupled) {
144 sim_num_cycles = sim_params["number_of_cardiac_cycles"].get_int64();
145 sim_pts_per_cycle =
146 sim_params["number_of_time_pts_per_cardiac_cycle"].get_int64();
147 sim_num_time_steps = (sim_pts_per_cycle - 1.0) * sim_num_cycles + 1.0;
148 sim_external_step_size = 0.0;
149 } else {
150 sim_num_cycles = 1;
151 sim_num_time_steps = sim_params["number_of_time_pts"].get_int64();
152 sim_pts_per_cycle = sim_num_time_steps;
153 try {
154 sim_external_step_size = sim_params["external_step_size"].get_double();
155 } catch (simdjson::simdjson_error) {
156 sim_external_step_size = 0.1;
157 }
158 }
159 sim_cardiac_cycle_period = -1.0; // Negative value indicates this has not
160 // been read from config file yet.
161 try {
162 sim_abs_tol = sim_params["absolute_tolerance"].get_double();
163 } catch (simdjson::simdjson_error) {
164 sim_abs_tol = 1e-8;
165 }
166 try {
167 sim_nliter = sim_params["maximum_nonlinear_iterations"].get_int64();
168 } catch (simdjson::simdjson_error) {
169 sim_nliter = 30;
170 }
171 try {
172 sim_steady_initial = sim_params["steady_initial"].get_bool();
173 } catch (simdjson::simdjson_error) {
174 sim_steady_initial = true;
175 }
176 try {
177 output_variable_based = sim_params["output_variable_based"].get_bool();
178 } catch (simdjson::simdjson_error) {
179 output_variable_based = false;
180 }
181 try {
182 output_interval = sim_params["output_interval"].get_int64();
183 } catch (simdjson::simdjson_error) {
184 output_interval = 1;
185 }
186 try {
187 output_mean_only = sim_params["output_mean_only"].get_bool();
188 } catch (simdjson::simdjson_error) {
189 output_mean_only = false;
190 }
191 try {
192 output_derivative = sim_params["output_derivative"].get_bool();
193 } catch (simdjson::simdjson_error) {
194 output_derivative = false;
195 }
196 try {
197 output_last_cycle_only = sim_params["output_last_cycle_only"].get_bool();
198 } catch (simdjson::simdjson_error) {
199 output_last_cycle_only = false;
200 }
201
202 // Create list to store block connections while generating blocks
203 std::vector<std::tuple<std::string_view, std::string_view>> connections;
204 int block_count = 0;
205
206 // Create vessels
207 std::map<std::int64_t, std::string_view> vessel_id_map;
208 for (auto vessel_config : config["vessels"]) {
209 auto vessel_values = vessel_config["zero_d_element_values"];
210 std::string_view vessel_name = vessel_config["vessel_name"].get_string();
211 vessel_id_map.insert({vessel_config["vessel_id"].get_int64(), vessel_name});
212 if (std::string_view(vessel_config["zero_d_element_type"].get_string()) ==
213 "BloodVessel") {
214 T R = vessel_values["R_poiseuille"].get_double();
215 T C;
216 try {
217 C = vessel_values["C"].get_double();
218 } catch (simdjson::simdjson_error) {
219 C = 0.0;
220 }
221 T L;
222 try {
223 L = vessel_values["L"].get_double();
224 } catch (simdjson::simdjson_error) {
225 L = 0.0;
226 }
227 T stenosis_coefficient;
228 try {
229 stenosis_coefficient =
230 vessel_values["stenosis_coefficient"].get_double();
231 } catch (simdjson::simdjson_error) {
232 stenosis_coefficient = 0.0;
233 }
234 model->blocks.push_back(new MODEL::BloodVessel<T>(
235 R = R, C = C, L = L, stenosis_coefficient = stenosis_coefficient,
236 static_cast<std::string>(vessel_name)));
237 model->block_index_map.insert(
238 {static_cast<std::string>(vessel_name), block_count});
239 block_count++;
240 DEBUG_MSG("Created vessel " << vessel_name);
241 } else {
242 throw std::invalid_argument("Unknown vessel type");
243 }
244
245 // Read connected boundary conditions
246 try {
247 auto inlet_bc = vessel_config["boundary_conditions"]["inlet"];
248 connections.push_back({inlet_bc.get_string(), vessel_name});
249 } catch (simdjson::simdjson_error) {
250 }
251
252 try {
253 auto outlet_bc = vessel_config["boundary_conditions"]["outlet"];
254 connections.push_back({vessel_name, outlet_bc.get_string()});
255 } catch (simdjson::simdjson_error) {
256 }
257 }
258
259 // Create map for boundary conditions to boundary condition type
260 std::map<std::string_view, std::string_view> bc_type_map;
261 for (auto bc_config : config["boundary_conditions"]) {
262 bc_type_map.insert({bc_config["bc_name"], bc_config["bc_type"]});
263 }
264 DEBUG_MSG("Created BC name->type map.");
265
266 try {
267 for (auto coupling_config : config["external_solver_coupling_blocks"]) {
268 std::string_view coupling_type = coupling_config["type"];
269 std::string_view coupling_name = coupling_config["name"];
270 std::string_view coupling_loc = coupling_config["location"];
271 bool periodic;
272 try {
273 periodic = coupling_config["periodic"].get_bool();
274 } catch (simdjson::simdjson_error) {
275 periodic = true;
276 }
277 auto coupling_values = coupling_config["values"];
278
279 // Create coupling block
280 std::vector<T> t_coupling;
281 try {
282 for (auto x : coupling_values["t"]) {
283 t_coupling.push_back(x.get_double());
284 }
285 } catch (simdjson::simdjson_error) {
286 t_coupling.push_back(0.0);
287 };
288
289 if (coupling_type == "FLOW") {
290 std::vector<T> Q_coupling;
291 try {
292 for (auto x : coupling_values["Q"].get_array()) {
293 Q_coupling.push_back(x.get_double());
294 }
295 } catch (simdjson::simdjson_error) {
296 Q_coupling.push_back(coupling_values["Q"].get_double());
297 }
298
299 MODEL::TimeDependentParameter q_coupling_param(t_coupling, Q_coupling,
300 periodic);
301 if ((q_coupling_param.isconstant == false) &&
302 (q_coupling_param.isperiodic == true)) {
303 if ((sim_cardiac_cycle_period > 0.0) &&
304 (q_coupling_param.cycle_period != sim_cardiac_cycle_period)) {
305 throw std::runtime_error(
306 "Inconsistent cardiac cycle period defined in "
307 "TimeDependentParameter");
308 } else {
309 sim_cardiac_cycle_period = q_coupling_param.cycle_period;
310 }
311 }
312 model->blocks.push_back(new MODEL::FlowReferenceBC<T>(
313 q_coupling_param, static_cast<std::string>(coupling_name),
314 static_cast<std::string>(coupling_loc)));
315 model->block_index_map.insert(
316 {static_cast<std::string>(coupling_name), block_count});
317 block_count++;
318 model->external_coupling_blocks.push_back(
319 static_cast<std::string>(coupling_name));
320 } else if (coupling_type == "PRESSURE") {
321 std::vector<T> P_coupling;
322 try {
323 for (auto x : coupling_values["P"].get_array()) {
324 P_coupling.push_back(x.get_double());
325 }
326 } catch (simdjson::simdjson_error) {
327 P_coupling.push_back(coupling_values["P"].get_double());
328 }
329 MODEL::TimeDependentParameter p_coupling_param(t_coupling, P_coupling,
330 periodic);
331 if ((p_coupling_param.isconstant == false) &&
332 (p_coupling_param.isperiodic == true)) {
333 if ((sim_cardiac_cycle_period > 0.0) &&
334 (p_coupling_param.cycle_period != sim_cardiac_cycle_period)) {
335 throw std::runtime_error(
336 "Inconsistent cardiac cycle period defined in "
337 "TimeDependentParameter");
338 } else {
339 sim_cardiac_cycle_period = p_coupling_param.cycle_period;
340 }
341 }
342 model->blocks.push_back(new MODEL::PressureReferenceBC<T>(
343 p_coupling_param, static_cast<std::string>(coupling_name),
344 static_cast<std::string>(coupling_loc)));
345 model->block_index_map.insert(
346 {static_cast<std::string>(coupling_name), block_count});
347 block_count++;
348 model->external_coupling_blocks.push_back(
349 static_cast<std::string>(coupling_name));
350 } else {
351 throw std::runtime_error(
352 "Error. Flowsolver coupling block types should be FLOW or "
353 "PRESSURE.");
354 }
355 DEBUG_MSG("Created coupling block " << coupling_name);
356
357 // Determine the type of connected block
358 std::string_view connected_block = coupling_config["connected_block"];
359 std::string_view connected_type;
360 int found_block = 0;
361 if (connected_block == "ClosedLoopHeartAndPulmonary") {
362 connected_type = "ClosedLoopHeartAndPulmonary";
363 found_block = 1;
364 } else {
365 try {
366 connected_type = bc_type_map.at(connected_block);
367 found_block = 1;
368 } catch (...) {
369 }
370 if (found_block == 0) {
371 // Search for connected_block in the list of vessel names
372 for (auto const vessel : vessel_id_map) {
373 if (connected_block == vessel.second) {
374 connected_type = "BloodVessel";
375 found_block = 1;
376 break;
377 }
378 }
379 }
380 if (found_block == 0) {
381 std::cout << "Error! Could not connected type for block: "
382 << connected_block << std::endl;
383 throw std::runtime_error("Terminating.");
384 }
385 } // connected_block != "ClosedLoopHeartAndPulmonary"
386 // Create connections
387 if (coupling_loc == "inlet") {
388 std::vector<std::string_view> possible_types = {
389 "RESISTANCE", "RCR", "ClosedLoopRCR",
390 "SimplifiedRCR", "CORONARY", "ClosedLoopCoronary",
391 "BloodVessel"};
392 if (std::find(std::begin(possible_types), std::end(possible_types),
393 connected_type) == std::end(possible_types)) {
394 throw std::runtime_error(
395 "Error: The specified connection type for inlet "
396 "external_coupling_block is invalid.");
397 }
398 connections.push_back({coupling_name, connected_block});
399 DEBUG_MSG("Created coupling block connection: " << coupling_name << "->"
400 << connected_block);
401 } else if (coupling_loc == "outlet") {
402 std::vector<std::string_view> possible_types = {
403 "ClosedLoopRCR", "ClosedLoopHeartAndPulmonary", "BloodVessel"};
404 if (std::find(std::begin(possible_types), std::end(possible_types),
405 connected_type) == std::end(possible_types)) {
406 throw std::runtime_error(
407 "Error: The specified connection type for outlet "
408 "external_coupling_block is invalid.");
409 }
410 // Add connection only for closedLoopRCR and BloodVessel. Connection to
411 // ClosedLoopHeartAndPulmonary will be handled in
412 // ClosedLoopHeartAndPulmonary creation.
413 if ((connected_type == "ClosedLoopRCR") ||
414 (connected_type == "BloodVessel")) {
415 connections.push_back({connected_block, coupling_name});
416 DEBUG_MSG("Created coupling block connection: "
417 << connected_block << "-> " << coupling_name);
418 } // connected_type == "ClosedLoopRCR"
419 } // coupling_loc
420 } // for (auto coupling_config : config["external_solver_coupling_blocks"])
421 } catch (simdjson::simdjson_error) {
422 }
423 DEBUG_MSG("Finished creating external coupling blocks.");
424
425 std::vector<std::string_view> closed_loop_bcs;
426
427 // Create boundary conditions
428 for (auto bc_config : config["boundary_conditions"]) {
429 std::string_view bc_type = bc_config["bc_type"];
430 std::string_view bc_name = bc_config["bc_name"];
431 auto bc_values = bc_config["bc_values"];
432
433 std::vector<T> t;
434 try {
435 for (auto x : bc_values["t"]) {
436 t.push_back(x.get_double());
437 }
438 } catch (simdjson::simdjson_error) {
439 t.push_back(0.0);
440 };
441
442 if (bc_type == "RCR") {
443 T Rp = bc_values["Rp"], C = bc_values["C"], Rd = bc_values["Rd"],
444 Pd = bc_values["Pd"];
445 model->blocks.push_back(new MODEL::WindkesselBC<T>(
446 Rp = Rp, C = C, Rd = Rd, Pd = Pd, static_cast<std::string>(bc_name)));
447 model->block_index_map.insert(
448 {static_cast<std::string>(bc_name), block_count});
449 block_count++;
450 DEBUG_MSG("Created boundary condition " << bc_name);
451 } else if (bc_type == "ClosedLoopRCR") {
452 T Rp = bc_values["Rp"], C = bc_values["C"], Rd = bc_values["Rd"];
453 bool closed_loop_outlet = bc_values["closed_loop_outlet"];
454 if (closed_loop_outlet == true) {
455 closed_loop_bcs.push_back(bc_name);
456 }
457 model->blocks.push_back(new MODEL::ClosedLoopRCRBC<T>(
458 Rp = Rp, C = C, Rd = Rd, closed_loop_outlet = closed_loop_outlet,
459 static_cast<std::string>(bc_name)));
460 model->block_index_map.insert(
461 {static_cast<std::string>(bc_name), block_count});
462 block_count++;
463 DEBUG_MSG("Created boundary condition " << bc_name);
464 } else if (bc_type == "FLOW") {
465 std::vector<T> Q;
466 try {
467 for (auto x : bc_values["Q"].get_array()) {
468 Q.push_back(x.get_double());
469 }
470 } catch (simdjson::simdjson_error) {
471 Q.push_back(bc_values["Q"].get_double());
472 }
473
474 MODEL::TimeDependentParameter q_param(t, Q);
475 if ((q_param.isconstant == false) && (q_param.isperiodic == true)) {
476 if ((sim_cardiac_cycle_period > 0.0) &&
477 (q_param.cycle_period != sim_cardiac_cycle_period)) {
478 throw std::runtime_error(
479 "Inconsistent cardiac cycle period defined in "
480 "TimeDependentParameter");
481 } else {
482 sim_cardiac_cycle_period = q_param.cycle_period;
483 }
484 }
485 model->blocks.push_back(new MODEL::FlowReferenceBC<T>(
486 q_param, static_cast<std::string>(bc_name)));
487 model->block_index_map.insert(
488 {static_cast<std::string>(bc_name), block_count});
489 block_count++;
490 DEBUG_MSG("Created boundary condition " << bc_name);
491
492 } else if (bc_type == "RESISTANCE") {
493 std::vector<T> R;
494 try {
495 for (auto x : bc_values["R"].get_array()) {
496 R.push_back(x.get_double());
497 }
498 } catch (simdjson::simdjson_error) {
499 R.push_back(bc_values["R"].get_double());
500 }
501 MODEL::TimeDependentParameter r_param(t, R);
502
503 std::vector<T> Pd;
504 try {
505 for (auto x : bc_values["Pd"].get_array()) {
506 Pd.push_back(x.get_double());
507 }
508 } catch (simdjson::simdjson_error) {
509 Pd.push_back(bc_values["Pd"].get_double());
510 }
511 MODEL::TimeDependentParameter pd_param(t, Pd);
512 model->blocks.push_back(new MODEL::ResistanceBC<T>(
513 r_param, pd_param, static_cast<std::string>(bc_name)));
514 model->block_index_map.insert(
515 {static_cast<std::string>(bc_name), block_count});
516 block_count++;
517 DEBUG_MSG("Created boundary condition " << bc_name);
518 } else if (bc_type == "PRESSURE") {
519 std::vector<T> P;
520 try {
521 for (auto x : bc_values["P"].get_array()) {
522 P.push_back(x.get_double());
523 }
524 } catch (simdjson::simdjson_error) {
525 P.push_back(bc_values["P"].get_double());
526 }
527 MODEL::TimeDependentParameter p_param(t, P);
528 if ((p_param.isconstant == false) && (p_param.isperiodic == true)) {
529 // if (p_param.isconstant == false) {
530 if ((sim_cardiac_cycle_period > 0.0) &&
531 (p_param.cycle_period != sim_cardiac_cycle_period)) {
532 throw std::runtime_error(
533 "Inconsistent cardiac cycle period defined in "
534 "TimeDependentParameter");
535 } else {
536 sim_cardiac_cycle_period = p_param.cycle_period;
537 }
538 }
539 model->blocks.push_back(new MODEL::PressureReferenceBC<T>(
540 p_param, static_cast<std::string>(bc_name)));
541 model->block_index_map.insert(
542 {static_cast<std::string>(bc_name), block_count});
543 block_count++;
544 DEBUG_MSG("Created boundary condition " << bc_name);
545 } else if (bc_type == "CORONARY") {
546 T Ra = bc_values["Ra1"], Ram = bc_values["Ra2"], Rv = bc_values["Rv1"],
547 Ca = bc_values["Ca"], Cim = bc_values["Cc"];
548 auto Pim_json = bc_values["Pim"], Pv_json = bc_values["P_v"];
549
550 std::vector<T> Pim;
551 try {
552 for (auto x : bc_values["Pim"].get_array()) {
553 Pim.push_back(x.get_double());
554 }
555 } catch (simdjson::simdjson_error) {
556 Pim.push_back(bc_values["Pim"].get_double());
557 }
558 MODEL::TimeDependentParameter pim_param(t, Pim);
559 std::vector<T> P_v;
560 try {
561 for (auto x : bc_values["P_v"].get_array()) {
562 P_v.push_back(x.get_double());
563 }
564 } catch (simdjson::simdjson_error) {
565 P_v.push_back(bc_values["P_v"].get_double());
566 }
567 MODEL::TimeDependentParameter pv_param(t, P_v);
568
569 model->blocks.push_back(new MODEL::OpenLoopCoronaryBC<T>(
570 Ra = Ra, Ram = Ram, Rv = Rv, Ca = Ca, Cim = Cim, pim_param, pv_param,
571 static_cast<std::string>(bc_name)));
572 model->block_index_map.insert(
573 {static_cast<std::string>(bc_name), block_count});
574 block_count++;
575 DEBUG_MSG("Created boundary condition " << bc_name);
576 } else if (bc_type == "ClosedLoopCoronary") {
577 T Ra = bc_values["Ra"], Ram = bc_values["Ram"], Rv = bc_values["Rv"],
578 Ca = bc_values["Ca"], Cim = bc_values["Cim"];
579 std::string_view side = bc_values["side"];
580 closed_loop_bcs.push_back(bc_name);
581 model->blocks.push_back(new MODEL::ClosedLoopCoronaryBC<T>(
582 Ra = Ra, Ram = Ram, Rv = Rv, Ca = Ca, Cim = Cim,
583 static_cast<std::string>(side), static_cast<std::string>(bc_name)));
584 model->block_index_map.insert(
585 {static_cast<std::string>(bc_name), block_count});
586 block_count++;
587 DEBUG_MSG("Created boundary condition " << bc_name);
588 } else {
589 throw std::invalid_argument("Unknown boundary condition type");
590 }
591 }
592
593 // Create junctions
594 for (auto junction_config : config["junctions"]) {
595 std::string_view j_type = junction_config["junction_type"].get_string();
596 std::string_view junction_name =
597 junction_config["junction_name"].get_string();
598 if ((j_type == "NORMAL_JUNCTION") || (j_type == "internal_junction")) {
599 model->blocks.push_back(
600 new MODEL::Junction<T>(static_cast<std::string>(junction_name)));
601 model->block_index_map.insert(
602 {static_cast<std::string>(junction_name), block_count});
603 block_count++;
604 } else if (j_type == "resistive_junction") {
605 std::vector<T> R;
606 for (auto x : junction_config["junction_values"]["R"].get_array()) {
607 R.push_back(x.get_double());
608 }
609 model->blocks.push_back(new MODEL::ResistiveJunction<T>(
610 R, static_cast<std::string>(junction_name)));
611 model->block_index_map.insert(
612 {static_cast<std::string>(junction_name), block_count});
613 block_count++;
614 } else if (j_type == "BloodVesselJunction") {
615 std::vector<T> R;
616 std::vector<T> C;
617 std::vector<T> L;
618 std::vector<T> stenosis_coefficient;
619 for (auto x :
620 junction_config["junction_values"]["R_poiseuille"].get_array()) {
621 R.push_back(x.get_double());
622 }
623 for (auto x : junction_config["junction_values"]["C"].get_array()) {
624 C.push_back(x.get_double());
625 }

638 block_count++;
639 } else {
640 throw std::invalid_argument("Unknown junction type");
641 }

643 // connections list
644 for (auto inlet_vessel : junction_config["inlet_vessels"]) {
645 connections.push_back(
646 {vessel_id_map[inlet_vessel.get_int64()], junction_name});
647 }
648 for (auto outlet_vessel : junction_config["outlet_vessels"]) {
649 connections.push_back(
650 {junction_name, vessel_id_map[outlet_vessel.get_int64()]});

652 DEBUG_MSG("Created junction " << junction_name);

658 ///< handling)

681 sim_cardiac_cycle_period = cycle_period;

684 // Convert to std::map to keep model blocks independent of simdjson
685 std::map<std::string, T> param_values;
686 param_values.insert(std::make_pair("Tsa", heart_params["Tsa"]));
687 param_values.insert(std::make_pair("tpwave", heart_params["tpwave"]));

791 // yet, set as default value of 1.0

802 // blocks For example, coronary block params that depend on heart block DOFs
803 for (auto &block : model->blocks) {
804 block->set_model_dependent_params(*model);
805 }

808 initial_state = ALGEBRA::State<T>::Zero(model->dofhandler.size());
809 // Initialize blocks that have fixed initial conditions
810 for (auto &block : model->blocks) {
811 block->set_ICs(initial_state);
812 }
813 try {

815 // Check for pressure_all or flow_all condition.
816 // This will initialize all pressure:* and flow:* variables.
817 T init_p, init_q;
818 bool init_p_flag = false;
819 bool init_q_flag = false;
820 try {
821 init_p = initial_condition["pressure_all"];
822 init_p_flag = true;
823 } catch (simdjson::simdjson_error) {
824 }
825 try {
826 init_q = initial_condition["flow_all"];
827 init_q_flag = true;
828 } catch (simdjson::simdjson_error) {
829 }
830 // Loop through variables and check for initial conditions.
831 for (size_t i = 0; i < model->dofhandler.size(); i++) {
832 try {
833 initial_state.y[i] = initial_condition[model->dofhandler.variables[i]];
834 } catch (simdjson::simdjson_error) {
835 std::string var_name = model->dofhandler.variables[i];
836 if ((init_p_flag == true) && ((var_name.substr(0, 9) == "pressure:") ||
837 (var_name.substr(0, 4) == "P_c:"))) {
838 initial_state.y[i] = init_p;
839 DEBUG_MSG("pressure_all initial condition for "
840 << model->dofhandler.variables[i]);
841 } else if ((init_q_flag == true) &&
842 (var_name.substr(0, 5) == "flow:")) {
843 initial_state.y[i] = init_q;
844 DEBUG_MSG("flow_all initial condition for "
845 << model->dofhandler.variables[i]);
846 } else {
847 DEBUG_MSG("No initial condition found for "
848 << model->dofhandler.variables[i]);
849 }
850 }
851 }
852 } catch (simdjson::simdjson_error) {
853 }
854 DEBUG_MSG("[configreader] END");
855}
