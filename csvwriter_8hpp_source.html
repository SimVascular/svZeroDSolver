46/**
47 * @brief Write results vessel based.
48 *
49 * @tparam T Scalar type (e.g. `float`, `double`)
50 * @param times Sequence of time steps corresponding to the solutions
51 * @param states Sequence of states corresponding to the time steps
52 * @param model The underlying model
53 * @param mean Toggle whether only the mean over all time steps should be
54 * written
55 * @return CSV encoded output string
56 */
57template <typename T>
58std::string to_vessel_csv(std::vector<T> &times,
59 std::vector<ALGEBRA::State<T>> &states,
60 MODEL::Model<T> &model, bool mean = false,
61 bool derivative = false) {
62 // Create string stream to buffer output
63 std::stringstream out;
64
65 // Create short and long buffer for lines
66 char sbuff[140];
67 char lbuff[236];
68
69 // Write column labels
70 if (derivative) {
71 out << "name,time,flow_in,flow_out,pressure_in,pressure_out,d_flow_in,d_"
72 "flow_out,d_pressure_in,d_pressure_out\n";
73 } else {
74 out << "name,time,flow_in,flow_out,pressure_in,pressure_out\n";
75 }
76
77 // Determine number of time steps
78 int num_steps = times.size();
79
80 unsigned int inflow_dof;
81 unsigned int outflow_dof;
82 unsigned int inpres_dof;
83 unsigned int outpres_dof;
84 for (size_t i = 0; i < model.get_num_blocks(); i++) {
85 auto block = model.get_block(i);
86 // Extract global solution indices of the block
87
88 if (dynamic_cast<const MODEL::BloodVessel<T> *>(block) == nullptr) {
89 continue;
90 }
91
92 std::string name = model.get_block_name(i);
93 inflow_dof = block->inlet_nodes[0]->flow_dof;
94 outflow_dof = block->outlet_nodes[0]->flow_dof;
95 inpres_dof = block->inlet_nodes[0]->pres_dof;
96 outpres_dof = block->outlet_nodes[0]->pres_dof;
97
98 // Write the solution of the block to the output file
99 if (derivative) {
100 if (mean) {
101 T inflow_mean = 0.0;
102 T outflow_mean = 0.0;
103 T inpres_mean = 0.0;
104 T outpres_mean = 0.0;
105 T d_inflow_mean = 0.0;
106 T d_outflow_mean = 0.0;
107 T d_inpres_mean = 0.0;
108 T d_outpres_mean = 0.0;
109 for (size_t i = 0; i < num_steps; i++) {
110 inflow_mean += states[i].y[inflow_dof];
111 outflow_mean += states[i].y[outflow_dof];
112 inpres_mean += states[i].y[inpres_dof];
113 outpres_mean += states[i].y[outpres_dof];
114 d_inflow_mean += states[i].ydot[inflow_dof];
115 d_outflow_mean += states[i].ydot[outflow_dof];
116 d_inpres_mean += states[i].ydot[inpres_dof];
117 d_outpres_mean += states[i].ydot[outpres_dof];
118 }
119 inflow_mean /= num_steps;
120 outflow_mean /= num_steps;
121 inpres_mean /= num_steps;
122 outpres_mean /= num_steps;
123 d_inflow_mean /= num_steps;
124 d_outflow_mean /= num_steps;
125 d_inpres_mean /= num_steps;
126 d_outpres_mean /= num_steps;
127 snprintf(
128 lbuff, 236, "%s,,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e\n",
129 name.c_str(), inflow_mean, outflow_mean, inpres_mean, outpres_mean,
130 d_inflow_mean, d_outflow_mean, d_inpres_mean, d_outpres_mean);
131 out << lbuff;
132 } else {
133 for (size_t i = 0; i < num_steps; i++) {
134 snprintf(lbuff, 236,
135 "%s,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e\n",
136 name.c_str(), times[i], states[i].y[inflow_dof],
137 states[i].y[outflow_dof], states[i].y[inpres_dof],
138 states[i].y[outpres_dof], states[i].ydot[inflow_dof],
139 states[i].ydot[outflow_dof], states[i].ydot[inpres_dof],
140 states[i].ydot[outpres_dof]);
141 out << lbuff;
142 }
143 }
144 } else {
145 if (mean) {
146 T inflow_mean = 0.0;
147 T outflow_mean = 0.0;
148 T inpres_mean = 0.0;
149 T outpres_mean = 0.0;
150 for (size_t i = 0; i < num_steps; i++) {
151 inflow_mean += states[i].y[inflow_dof];
152 outflow_mean += states[i].y[outflow_dof];
153 inpres_mean += states[i].y[inpres_dof];
154 outpres_mean += states[i].y[outpres_dof];
155 }
156 inflow_mean /= num_steps;
157 outflow_mean /= num_steps;
158 inpres_mean /= num_steps;
159 outpres_mean /= num_steps;
160 snprintf(sbuff, 140, "%s,,%.16e,%.16e,%.16e,%.16e\n", name.c_str(),
161 inflow_mean, outflow_mean, inpres_mean, outpres_mean);
162 out << sbuff;
163 } else {
164 for (size_t i = 0; i < num_steps; i++) {
165 snprintf(sbuff, 140, "%s,%.16e,%.16e,%.16e,%.16e,%.16e\n",
166 name.c_str(), times[i], states[i].y[inflow_dof],
167 states[i].y[outflow_dof], states[i].y[inpres_dof],
168 states[i].y[outpres_dof]);
169 out << sbuff;
170 }
171 }
172 }
173 }
174
175 return out.str();
176}

178/**
179 * @brief Write results variable based.
180 *
181 * @tparam T Scalar type (e.g. `float`, `double`)
182 * @param times Sequence of time steps corresponding to the solutions
183 * @param states Sequence of states corresponding to the time steps
184 * @param model The underlying model
185 * @param mean Toggle whether only the mean over all time steps should be
186 * written
187 * @return CSV encoded output string
188 */
189template <typename T>
190std::string to_variable_csv(std::vector<T> &times,
191 std::vector<ALGEBRA::State<T>> &states,
192 MODEL::Model<T> &model, bool mean = false,
193 bool derivative = false) {
194 // Create string stream to buffer output
195 std::stringstream out;
196
197 // Create short and long buffer for lines
198 char sbuff[87];
199 char lbuff[110];
200
201 // Determine number of time steps
202 int num_steps = times.size();
203
204 // Write column labels
205 if (derivative) {
206 out << "name,time,y,ydot\n";
207 if (mean) {
208 for (size_t i = 0; i < model.dofhandler.size(); i++) {
209 std::string name = model.dofhandler.variables[i];
210 T mean_y = 0.0;
211 T mean_ydot = 0.0;
212 for (size_t j = 0; j < num_steps; j++) {
213 mean_y += states[j].y[i];
214 mean_ydot += states[j].ydot[i];
215 }
216 mean_y /= num_steps;
217 mean_ydot /= num_steps;
218 snprintf(lbuff, 110, "%s,,%.16e,%.16e\n", name.c_str(), mean_y,
219 mean_ydot);
220 out << lbuff;
221 }
222 } else {
223 for (size_t i = 0; i < model.dofhandler.size(); i++) {
224 std::string name = model.dofhandler.variables[i];
225 for (size_t j = 0; j < num_steps; j++) {
226 snprintf(lbuff, 110, "%s,%.16e,%.16e,%.16e\n", name.c_str(), times[j],
227 states[j].y[i], states[j].ydot[i]);
228 out << lbuff;
229 }
230 }
231 }
232 } else {
233 out << "name,time,y\n";
234 if (mean) {
235 for (size_t i = 0; i < model.dofhandler.size(); i++) {
236 std::string name = model.dofhandler.variables[i];
237 T mean_y = 0.0;
238 for (size_t j = 0; j < num_steps; j++) {
239 mean_y += states[j].y[i];
240 }
241 mean_y /= num_steps;
242 snprintf(sbuff, 87, "%s,,%.16e\n", name.c_str(), mean_y);
243 out << sbuff;
244 }
245 } else {
246 for (size_t i = 0; i < model.dofhandler.size(); i++) {
247 std::string name = model.dofhandler.variables[i];
248 for (size_t j = 0; j < num_steps; j++) {
249 snprintf(sbuff, 87, "%s,%.16e,%.16e\n", name.c_str(), times[j],
250 states[j].y[i]);
251 out << sbuff;
252 }
253 }
254 }
255 }
256
257 return out.str();
258}
