14 /**
15 * @brief Junction
16 *
17 * Models a junction with arbitrary inlets and outlets. Across all inlets and
18 * outlets of the junction, mass is conserved and pressure is continuous.
19 *
20 * \f[
21 * \begin{circuitikz}
22 * \draw node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
23 * \draw (1,0) node[anchor=south]{$P_{in}$} to [short, *-*] (3.0,0);
24 * \draw (3,0) node[anchor=south]{} to [short, -*] (4.5,1.0);
25 * \draw (4.3,1.1) node[anchor=south] {$P_{out,1}$};
26 * \draw (3,0) node[anchor=south]{} to [short, -*] (4.5,-1.0);
27 * \draw (4.3,-1.1) node[anchor=north] {$P_{out,2}$};
28 * \draw [-latex] (4.65,1.1) -- (5.25,1.5) node[right] {$Q_{out,1}$};
29 * \draw [-latex] (4.65,-1.1) -- (5.25,-1.5) node[right] {$Q_{out,2}$};
30 * \end{circuitikz}
31 * \f]
32 *
33 * ### Governing equations
34 *
35 * \f[
36 * \sum_{i}^{n_{inlets}} Q_{in, i}=\sum_{j}^{n_{outlets}} Q_{out, j}
37 * \f]
38 *
39 * \f[
40 * P_{i}=P_{j} \quad \mathrm{with} \quad i \neq j
41 * \f]
42 *
43 * ### Local contributions
44 *
45 * \f[
46 * \mathbf{y}^{e}=\left[\begin{array}{llllllllll}P_{in, 1}^{e} & Q_{in, 1}^{e} & \dots & P_{in, i}^{e} & Q_{in, i}^{e} & P_{out, 1}^{e} & Q_{out, 1}^{e} & \dots & P_{out, i}^{e} & Q_{out, i}^{e}\end{array}\right]
47 * \f]
48 *
49 * Mass conservation
50 *
51 * \f[
52 * \mathbf{F}^{e}_1 = \left[\begin{array}{llllllllll}0 & 1 & 0 & 1 & \dots & 0 & -1 & 0 & -1 & \dots\end{array}\right]
53 * \f]
54 *
55 * Due to the pressure continuity, we can write for all independent pressure pairs:
56 * \f[
57 * \mathbf{F}^{e}_{2,...,n} = \left[\begin{array}{lllll}\dots & \underbrace{1}_{P_i} & \dots & \underbrace{1}_{P_j} & \dots\end{array}\right] \quad \mathrm{with} \quad i \neq j
58 * \f]
59 *
60 * @tparam T Scalar type (e.g. `float`, `double`)
61 */

152 template <typename T>
153 void Junction<T>::update_constant(ALGEBRA::DenseSystem<T> &system)
154 {
155 // Continuous pressure condition
156 for (size_t i = 0; i < (num_inlets + num_outlets - 1); i++)
157 {
158 system.F(this->global_eqn_ids[i], this->global_var_ids[0]) = 1.0;
159 system.F(this->global_eqn_ids[i], this->global_var_ids[2 * i + 2]) = -1.0;
160 num_triplets["F"] += 2;
161 }
162
163 // Conservation of mass
164 for (size_t i = 1; i < num_inlets * 2; i = i + 2)
165 {
166 system.F(this->global_eqn_ids[num_inlets + num_outlets - 1], this->global_var_ids[i]) = 1.0;
167 num_triplets["F"] += 1;
168 }
169 for (size_t i = (num_inlets * 2) + 1; i < (num_inlets + num_outlets) * 2; i = i + 2)
170 {
171 system.F(this->global_eqn_ids[num_inlets + num_outlets - 1], this->global_var_ids[i]) = -1.0;
172 num_triplets["F"] += 1;
173 }
174 }

182 system.F.coeffRef(this->global_eqn_ids[i], this->global_var_ids[2 * i + 2]) = -1.0;
183 }
184 for (size_t i = 1; i < num_inlets * 2; i = i + 2)
185 {
186 system.F.coeffRef(this->global_eqn_ids[num_inlets + num_outlets - 1], this->global_var_ids[i]) = 1.0;
187 }
188 for (size_t i = (num_inlets * 2) + 1; i < (num_inlets + num_outlets) * 2; i = i + 2)
189 {
190 system.F.coeffRef(this->global_eqn_ids[num_inlets + num_outlets - 1], this->global_var_ids[i]) = -1.0;
