119template <typename T>
120SparseSystem<T>::SparseSystem() {}

122template <typename T>
123SparseSystem<T>::SparseSystem(unsigned int n) {
124 F = Eigen::SparseMatrix<T>(n, n);
125 E = Eigen::SparseMatrix<T>(n, n);
126 D = Eigen::SparseMatrix<T>(n, n);
127 C = Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n);
128
129 jacobian = Eigen::SparseMatrix<T>(n, n);
130 residual = Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n);
131 dy = Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n);
132}

134template <typename T>
135SparseSystem<T>::~SparseSystem() {}

137template <typename T>
138void SparseSystem<T>::reserve(MODEL::Model<T> &model) {
139 auto num_triplets = model.get_num_triplets();
140 F.reserve(num_triplets["F"]);
141 E.reserve(num_triplets["E"]);
142 D.reserve(num_triplets["D"]);
143 model.update_constant(*this);
144 model.update_time(*this, 0.0);
145 Eigen::Matrix<T, Eigen::Dynamic, 1> dummy_y =
146 Eigen::Matrix<T, Eigen::Dynamic, 1>::Ones(residual.size());
147 model.update_solution(*this, dummy_y);
148 F.makeCompressed();
149 E.makeCompressed();
150 D.makeCompressed();
151 jacobian.reserve(num_triplets["F"] + num_triplets["E"]); // Just an estimate
152 update_jacobian(1.0); // Update it once to have sparsity pattern
153 jacobian.makeCompressed();
154 solver->analyzePattern(jacobian); // Let solver analyze pattern
155}

157template <typename T>
158void SparseSystem<T>::update_residual(
159 Eigen::Matrix<T, Eigen::Dynamic, 1> &y,
160 Eigen::Matrix<T, Eigen::Dynamic, 1> &ydot) {
161 residual.setZero();
162 residual -= C;
163 residual.noalias() -= E * ydot;
164 residual.noalias() -= F * y;
165}
166
167template <typename T>
168void SparseSystem<T>::update_jacobian(T e_coeff) {
169 jacobian.setZero();
170 jacobian += F + D + E * e_coeff;
171}
172
173template <typename T>
174void SparseSystem<T>::solve() {
175 solver->factorize(jacobian);
176 dy.setZero();
177 dy += solver->solve(residual);
178}
