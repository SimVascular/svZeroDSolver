129template <typename T>
130Solver<T>::Solver(const nlohmann::json& config) {
131 DEBUG_MSG("Read simulation parameters");
132 simparams = IO::load_simulation_params<T>(config);
133 DEBUG_MSG("Load model");
134 model = MODEL::Model<T>();
135 IO::load_simulation_model<T>(config, model);
136 DEBUG_MSG("Load initial condition");
137 initial_state = IO::load_initial_condition<T>(config, model);
138
139 DEBUG_MSG("Cardiac cycle period " << model.cardiac_cycle_period);
140
141 // Calculate time step size
142 if (!simparams.sim_coupled) {
143 simparams.sim_time_step_size =
144 model.cardiac_cycle_period / (T(simparams.sim_pts_per_cycle) - 1.0);
145 } else {
146 simparams.sim_time_step_size = simparams.sim_external_step_size /
147 (T(simparams.sim_num_time_steps) - 1.0);
148 }
149 sanity_checks();
150}

155template <typename T>
156void Solver<T>::run() {
157 auto state = initial_state;
158
159 // Create steady initial
160 if (simparams.sim_steady_initial) {
161 DEBUG_MSG("Calculate steady initial condition");
162 T time_step_size_steady = model.cardiac_cycle_period / 10.0;
163 model.to_steady();
164 ALGEBRA::Integrator<T> integrator_steady(&model, time_step_size_steady, 0.1,
165 simparams.sim_abs_tol,
166 simparams.sim_nliter);
167 for (int i = 0; i < 31; i++) {
168 state = integrator_steady.step(state, time_step_size_steady * T(i));
169 }
170 model.to_unsteady();
171 }
172
173 // Set-up integrator
174 DEBUG_MSG("Setup time integration");
175 ALGEBRA::Integrator<T> integrator(&model, simparams.sim_time_step_size, 0.1,
176 simparams.sim_abs_tol,
177 simparams.sim_nliter);
178
179 // Initialize loop
180 states = std::vector<ALGEBRA::State<T>>();
181 times = std::vector<T>();
182 if (simparams.output_all_cycles) {
183 int num_states =
184 simparams.sim_num_time_steps / simparams.output_interval + 1;
185 states.reserve(num_states);
186 times.reserve(num_states);
187 } else {
188 int num_states =
189 simparams.sim_pts_per_cycle / simparams.output_interval + 1;
190 states.reserve(num_states);
191 times.reserve(num_states);
192 }
193 T time = 0.0;
194
195 // Run integrator
196 DEBUG_MSG("Run time integration");
197 int interval_counter = 0;
198 int start_last_cycle =
199 simparams.sim_num_time_steps - simparams.sim_pts_per_cycle;
200 if (simparams.output_all_cycles || (0 >= start_last_cycle)) {
201 times.push_back(time);
202 states.push_back(std::move(state));
203 }
204 for (int i = 1; i < simparams.sim_num_time_steps; i++) {
205 state = integrator.step(state, time);
206 interval_counter += 1;
207 time = simparams.sim_time_step_size * T(i);
208 if ((interval_counter == simparams.output_interval) ||
209 (!simparams.output_all_cycles && (i == start_last_cycle))) {
210 if (simparams.output_all_cycles || (i >= start_last_cycle)) {
211 times.push_back(time);
212 states.push_back(std::move(state));
213 }
214 interval_counter = 0;
215 }
216 }
217
218 // Make times start from 0
219 if (!simparams.output_all_cycles) {
220 T start_time = times[0];
221 for (auto& time : times) {
222 time -= start_time;
223 }
224 }
225}

232template <typename T>
233std::string Solver<T>::get_full_result() {
234 std::string output;
235 if (simparams.output_variable_based) {
236 output =
237 IO::to_variable_csv<T>(times, states, model, simparams.output_mean_only,
238 simparams.output_derivative);
239 } else {
240 output =
241 IO::to_vessel_csv<T>(times, states, model, simparams.output_mean_only,
242 simparams.output_derivative);
243 }
244 return output;
245}

247template <typename T>
248Eigen::VectorXd Solver<T>::get_single_result(std::string dof_name) {
249 int dof_index = model.dofhandler.get_variable_index(dof_name);
250 int num_states = states.size();
251 Eigen::VectorXd result = Eigen::VectorXd::Zero(num_states);
252 for (size_t i = 0; i < num_states; i++) result[i] = states[i].y[dof_index];
253 return result;
254}

256template <typename T>
257T Solver<T>::get_single_result_avg(std::string dof_name) {
258 int dof_index = model.dofhandler.get_variable_index(dof_name);
259 int num_states = states.size();
260 Eigen::VectorXd result = Eigen::VectorXd::Zero(num_states);
261 for (size_t i = 0; i < num_states; i++) result[i] = states[i].y[dof_index];
262 return result.mean();
263}

265template <typename T>
266void Solver<T>::update_block_params(std::string block_name,
267 std::vector<T> new_params) {
268 auto block = model.get_block(block_name);
269 if (new_params.size() != block->global_param_ids.size()) {
270 throw std::runtime_error(
271 "Parameter update failed! Number of provided parameters does not match "
272 "with block parameters.");
273 }
274 for (size_t i = 0; i < new_params.size(); i++) {
275 model.get_parameter(block->global_param_ids[i])->update(new_params[i]);
276 }
277}

279template <typename T>
280void Solver<T>::sanity_checks() {
281 // Check that steady initial is not used with ClosedLoopHeartAndPulmonary
282 if ((simparams.sim_steady_initial == true) &&
283 (model.get_block("CLH") != nullptr)) {
284 std::runtime_error(
285 "ERROR: Steady initial condition is not compatible with "
286 "ClosedLoopHeartAndPulmonary block.");
287 }
288}
